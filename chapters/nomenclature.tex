%% First, define the different groups you want in the nomenclature
\renewcommand\nomgroup[1]{%
  \item[\bfseries%
  \ifstrequal{#1}{S}{Symbols}{%
  \ifstrequal{#1}{M}{Matrix/tensors}{%
  \ifstrequal{#1}{N}{Non-dimensional group}{%
  \ifstrequal{#1}{A}{Acronyms}{%
  \ifstrequal{#1}{F}{Functions/operators}{}}}}}%
]}
\setlength{\nomitemsep}{-\parsep}
\makenomenclature%

% Symbols (S)
\nomenclature[S]{$\vector{v}$}{Vorticity (local, instantaneous)}
\nomenclature[S]{$\vector{u}$}{Velocity (local, instantaneous)}
\nomenclature[S]{$t$}{Time}
\nomenclature[S]{$x$}{Spatial dimension}

% Functions/operators
\nomenclature[F]{$\triangle$}{Vector Laplacian operator}
\nomenclature[F]{$\vert\vert\ast\vert\vert_2^2$}{$\ell_2$ norm}

% Nondimensional numbers (N)
\nomenclature[N]{$\sym{Re}$}{Reynolds Number}
\nomenclature[N]{$\sym{St}$}{Strouhal number}

% Acronyms (A)
\nomenclature[A]{MSE}{Mean Squared Error}

% Matrix/Tensors (M)
\nomenclature[M]{$\mat{\Phi}$}{POD modes}
\nomenclature[M]{$\mat{U}$}{Dataset containing only the velocity}

\printnomenclature%
